// Herança

using System;

class Pessoa
{
    public string nome;
    public int idade;

    public void Apresentar()
    {
        Console.WriteLine($"Nome: {nome} Idade: {idade}");
    }
}


class Aluno : Pessoa
{
    public string Curso;
}


class Program
{
    static void Main()
    {
        {
            Aluno a = new Aluno();

            a.nome = "Carlos";
            a.idade = 27;
            a.Curso = "Software Engeenier";

            a.Apresentar();
        }
    }
}
